47 # binary GCD (aka Stein's) algorithm
48 # about 1.7x (2.1x) faster for random Int64s (Int128s)
49 function gcd(a::T, b::T) where T<:BitInteger
50 a == 0 && return checked_abs(b)
51 b == 0 && return checked_abs(a)
52 za = trailing_zeros(a)
53 zb = trailing_zeros(b)
54 k = min(za, zb)
55 u = unsigned(abs(a >> za))
56 v = unsigned(abs(b >> zb))
57 while u != v
58 if u > v
59 u, v = v, u
60 end
61 v -= u
62 v >>= trailing_zeros(v)
63 end
64 r = u << k
65 # T(r) would throw InexactError; we want OverflowError instead
66 r > typemax(T) && __throw_gcd_overflow(a, b)
67 r % T
68 end
69 @noinline __throw_gcd_overflow(a, b) = throw(OverflowError("gcd($a, $b) overflows"))

141 gcdx(x,y)
142
143 Computes the greatest common (positive) divisor of `x` and `y` and their Bézout
144 coefficients, i.e. the integer coefficients `u` and `v` that satisfy
145 ``ux+vy = d = gcd(x,y)``. ``gcdx(x,y)`` returns ``(d,u,v)``.
146
147 The arguments may be integer and rational numbers.
148
149 !!! compat "Julia 1.4"
150 Rational arguments require Julia 1.4 or later.
151
152 # Examples
153 ```jldoctest
154 julia> gcdx(12, 42)
155 (6, -3, 1)
156
157 julia> gcdx(240, 46)
158 (2, -9, 47)
159 ```
160
161 !!! note
162 Bézout coefficients are *not* uniquely defined. `gcdx` returns the minimal
163 Bézout coefficients that are computed by the extended Euclidean algorithm.
164 (Ref: D. Knuth, TAoCP, 2/e, p. 325, Algorithm X.)
165 For signed integers, these coefficients `u` and `v` are minimal in
166 the sense that ``|u| < |y/d|`` and ``|v| < |x/d|``. Furthermore,
167 the signs of `u` and `v` are chosen so that `d` is positive.
168 For unsigned integers, the coefficients `u` and `v` might be near
169 their `typemax`, and the identity then holds only via the unsigned
170 integers' modulo arithmetic.
171 """
172 function gcdx(a::U, b::V) where {U<:Integer, V<:Integer}
173 T = promote_type(U, V)
174 # a0, b0 = a, b
175 s0, s1 = oneunit(T), zero(T)
176 t0, t1 = s1, s0
177 # The loop invariant is: s0*a0 + t0*b0 == a
178 x = a % T
179 y = b % T
180 while y != 0
181 q = div(x, y)
182 x, y = y, rem(x, y)
183 s0, s1 = s1, s0 - q*s1
184 t0, t1 = t1, t0 - q*t1
185 end
186 x < 0 ? (-x, -s0, -t0) : (x, s0, t0)
187 end

342 function powermod(x::Integer, p::Integer, m::T) where T<:Integer
343 p < 0 && return powermod(invmod(x, m), -p, m)
344 p == 0 && return mod(one(m),m)
345 (m == 1 || m == -1) && return zero(m)
346 b = oftype(m,mod(x,m)) # this also checks for divide by zero
347
348 t = prevpow(2, p)
349 r::T = 1
350 while true
351 if p >= t
352 r = mod(widemul(r,b),m)
353 p -= t
354 end
355 t >>>= 1
356 t <= 0 && break
357 r = mod(widemul(r,r),m)
358 end
359 return r
360 end

665 x = div(x,b)
666 else
667 a[i] = digits[1+mod(x,-b)]
668 x = cld(x,b)
669 end
670 i -= 1
671 end
672 if neg; a[1]='-'; end
673 String(a)
674 end

781 """
782 digits!(array, n::Integer; base::Integer = 10)
783
784 Fills an array of the digits of `n` in the given base. More significant digits are at higher
785 indices. If the array length is insufficient, the least significant digits are filled up to
786 the array length. If the array length is excessive, the excess portion is filled with zeros.
787
788 # Examples
789 ```jldoctest
790 julia> digits!([2,2,2,2], 10, base = 2)
791 4-element Array{Int64,1}:
792 0
793 1
794 0
795 1
796
797 julia> digits!([2,2,2,2,2,2], 10, base = 2)
798 6-element Array{Int64,1}:
799 0
800 1
801 0
802 1
803 0
804 0
805 ```
806 """
807 function digits!(a::AbstractVector{T}, n::Integer; base::Integer = 10) where T<:Integer
808 2 <= abs(base) || throw(DomainError(base, "base must be ≥ 2 or ≤ -2"))
809 hastypemax(T) && abs(base) - 1 > typemax(T) &&
810 throw(ArgumentError("type $T too small for base $base"))
811 isempty(a) && return a
812
813 if base > 0
814 if ispow2(base) && n >= 0 && n isa Base.BitInteger && base <= typemax(Int)
815 base = Int(base)
816 k = trailing_zeros(base)
817 c = base - 1
818 for i in eachindex(a)
819 a[i] = (n >> (k * (i - firstindex(a)))) & c
820 end
821 else
822 for i in eachindex(a)
823 n, d = divrem(n, base)
824 a[i] = d
825 end
826 end
827 else
828 # manually peel one loop iteration for type stability
829 n, d = fldmod(n, -base)
830 a[firstindex(a)] = d
831 n = -signed(n)
832 for i in firstindex(a)+1:lastindex(a)
833 n, d = fldmod(n, -base)
834 a[i] = d
835 n = -n
836 end
837 end
838 return a
839 end

862 """
863 factorial(n::Integer)
864
865 Factorial of `n`. If `n` is an [`Integer`](@ref), the factorial is computed as an
866 integer (promoted to at least 64 bits). Note that this may overflow if `n` is not small,
867 but you can use `factorial(big(n))` to compute the result exactly in arbitrary precision.
868
869 # Examples
870 ```jldoctest
871 julia> factorial(6)
872 720
873
874 julia> factorial(21)
875 ERROR: OverflowError: 21 is too large to look up in the table; consider using `factorial(big(21))` instead
876 Stacktrace:
877 [...]

880 51090942171709440000
881 ```
882
883 # See also
884 * [`binomial`](@ref)
885
886 # External links
887 * [Factorial](https://en.wikipedia.org/wiki/Factorial) on Wikipedia.
888 """
889 function factorial(n::Integer)
890 n < 0 && throw(DomainError(n, "`n` must be nonnegative."))
891 f::typeof(n*n) = 1
892 for i::typeof(n*n) = 2:n
893 f *= i
894 end
895 return f
896 end

898 """
899 binomial(n::Integer, k::Integer)
900
901 The _binomial coefficient_ ``\\binom{n}{k}``, being the coefficient of the ``k``th term in
902 the polynomial expansion of ``(1+x)^n``.
903
904 If ``n`` is non-negative, then it is the number of ways to choose `k` out of `n` items:
905 ```math
906 \\binom{n}{k} = \\frac{n!}{k! (n-k)!}
907 ```
908 where ``n!`` is the [`factorial`](@ref) function.
909
910 If ``n`` is negative, then it is defined in terms of the identity
911 ```math
912 \\binom{n}{k} = (-1)^k \\binom{k-n-1}{k}
913 ```
914
915 # Examples
916 ```jldoctest
917 julia> binomial(5, 3)
918 10
919
920 julia> factorial(5) ÷ (factorial(5-3) * factorial(3))
921 10
922
923 julia> binomial(-5, 3)
924 -35
925 ```
926
927 # See also
928 * [`factorial`](@ref)
929
930 # External links
931 * [Binomial coefficient](https://en.wikipedia.org/wiki/Binomial_coefficient) on Wikipedia.
932 """
933 function binomial(n::T, k::T) where T<:Integer
934 n0, k0 = n, k
935 k < 0 && return zero(T)
936 sgn = one(T)
937 if n < 0
938 n = -n + k -1
939 if isodd(k)
940 sgn = -sgn
941 end
942 end
943 k > n && return zero(T)
944 (k == 0 || k == n) && return sgn
945 k == 1 && return sgn*n
946 if k > (n>>1)
947 k = (n - k)
948 end
949 x::T = nn = n - k + 1
950 nn += 1
951 rr = 2
952 while rr <= k
953 xt = div(widemul(x, nn), rr)
954 x = xt % T
955 x == xt || throw(OverflowError("binomial($n0, $k0) overflows"))
956 rr += 1
957 nn += 1
958 end
959 convert(T, copysign(x, sgn))
960 end
