139 function skew_expm(B::Mat)::Mat
140 """Expm for skew symmetric matrices.
141
142 Summary:
143 matrix exponential for skew symmetric matrices. about 40% faster than
144 the standard exp.jl function. This function can be used to take the
145 skew symmetric skew_from_vec matrix of an axis angle vector, and
146 producing the orthogonal Direction Cosine Matrix (DCM)
147 skew_from_vec corresponds.
148
149 Args:
150 B: skew symmetric matrix :: AbstractArray{Float64,2}
151
152 Returns:
153 orthognal matrix :: AbstractArray{Float64,2}
154 """
155
156 # axis (from axis-angle)
157 phi = vec_from_skew(B)
158
159 # angle (from axis-angle)
160 theta = norm(phi)
161
162 # axis
163 if theta == 0
164 r = [0.0; 0.0; 0.0]
165 else
166 r = phi / theta
167 end
168
169 # closed form skew symmetric matrix exponential
170 return (I + sin(theta) * skew_from_vec(r) + (1.0 - cos(theta)) *
171 skew_from_vec(r) * skew_from_vec(r))
172 end

176 function ortho_logm(Q::Mat)::Mat
177 """Matrix logarithm for 3x3 orthogonal matrices (like DCM's).
178
179 Summary:
180 This is both faster and more robust than log.jl (180 degree rotations)
181
182 Args:
183 Q: orthogonal matrix (like a DCM) :: AbstractArray{Float64,2}
184
185 Returns:
186 skew symmetric matrix :: AbstractArray{Float64,2}
187 """
188
189 val = (tr(Q) - 1) / 2
190
191 if abs(val - 1) < 1e-10
192 # no rotation
193 phi = [0.0; 0.0; 0.0]
194 elseif abs(val + 1) < 1e-10
195 # 180 degree rotation
196 M = I + Q
197 r = M[1, :] / norm(M[1, :])
198 theta = pi
199 phi = r * theta
200 else
201 # normal rotation (0-180 degrees)
202 theta = acos(val)
203 r = -(1 / (2 * sin(theta))) *
204 [Q[2, 3] - Q[3, 2]; Q[3, 1] - Q[1, 3]; Q[1, 2] - Q[2, 1]]
205 phi = r * theta
206 end
207
208 return skew_from_vec(phi)
209 end

319 function dcm_from_q(q::Vec)::Mat
320 """DCM from quaternion, hamilton product, scalar last"""
321
322 # pull our the parameters from the quaternion
323 q1,q2,q3,q4 = normalize(q)
324
325 # DCM
326 Q = [(2*q1^2+2*q4^2-1) 2*(q1*q2 - q3*q4) 2*(q1*q3 + q2*q4);
327 2*(q1*q2 + q3*q4) (2*q2^2+2*q4^2-1) 2*(q2*q3 - q1*q4);
328 2*(q1*q3 - q2*q4) 2*(q2*q3 + q1*q4) (2*q3^2+2*q4^2-1)]

375 function q_from_dcm(dcm::Mat)::Vec
376 """Kane/Levinson convention, scalar last"""
377 R = dcm
378 T = R[1,1] + R[2,2] + R[3,3]
379 if T> R[1,1] && T > R[2,2] && T>R[3,3]
380 q4 = .5*sqrt(1+T)
381 r = .25/q4
382 q1 = (R[3,2] - R[2,3])*r
383 q2 = (R[1,3] - R[3,1])*r
384 q3 = (R[2,1] - R[1,2])*r
385 elseif R[1,1]>R[2,2] && R[1,1]>R[3,3]
386 q1 = .5*sqrt(1-T + 2*R[1,1])
387 r = .25/q1
388 q4 = (R[3,2] - R[2,3])*r
389 q2 = (R[1,2] + R[2,1])*r
390 q3 = (R[1,3] + R[3,1])*r
391 elseif R[2,2]>R[3,3]
392 q2 = .5*sqrt(1-T + 2*R[2,2])
393 r = .25/q2
394 q4 = (R[1,3] - R[3,1])*r
395 q1 = (R[1,2] + R[2,1])*r
396 q3 = (R[2,3] + R[3,2])*r
397 else
398 q3 = .5*sqrt(1-T + 2*R[3,3])
399 r = .25/q3
400 q4 = (R[2,1] - R[1,2])*r
401 q1 = (R[1,3] + R[3,1])*r
