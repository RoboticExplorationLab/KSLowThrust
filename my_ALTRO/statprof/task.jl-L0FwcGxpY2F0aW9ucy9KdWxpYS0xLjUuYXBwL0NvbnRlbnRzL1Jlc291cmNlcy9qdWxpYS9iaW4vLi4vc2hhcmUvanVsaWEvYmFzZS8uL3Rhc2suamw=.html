28 """
29 CompositeException
30
31 Wrap a `Vector` of exceptions thrown by a [`Task`](@ref) (e.g. generated from a remote worker over a channel
32 or an asynchronously executing local I/O write or a remote worker under `pmap`) with information about the series of exceptions.
33 For example, if a group of workers are executing several tasks, and multiple workers fail, the resulting `CompositeException` will
34 contain a "bundle" of information from each worker indicating where and why the exception(s) occurred.
35 """

38 CompositeException() = new(Any[])
39 CompositeException(exceptions) = new(exceptions)
40 end
41 length(c::CompositeException) = length(c.exceptions)
42 push!(c::CompositeException, ex) = push!(c.exceptions, ex)
43 isempty(c::CompositeException) = isempty(c.exceptions)
44 iterate(c::CompositeException, state...) = iterate(c.exceptions, state...)
45 eltype(::Type{CompositeException}) = Any

213 """
214 task_local_storage(body, key, value)
215
216 Call the function `body` with a modified task-local storage, in which `value` is assigned to
217 `key`; the previous value of `key`, or lack thereof, is restored afterwards. Useful
218 for emulating dynamic scoping.
219 """
220 function task_local_storage(body::Function, key, val)
221 tls = task_local_storage()
222 hadkey = haskey(tls, key)
223 old = get(tls, key, nothing)
224 tls[key] = val
225 try
226 return body()
227 finally
228 hadkey ? (tls[key] = old) : delete!(tls, key)
229 end

289 function sync_end(c::Channel{Any})
290 local c_ex
291 while isready(c)
292 r = take!(c)
293 if isa(r, Task)
294 _wait(r)
295 if istaskfailed(r)
296 if !@isdefined(c_ex)
297 c_ex = CompositeException()
298 end
299 push!(c_ex, TaskFailedException(r))
300 end
301 else
302 try
303 wait(r)
304 catch e
305 if !@isdefined(c_ex)
306 c_ex = CompositeException()
307 end
308 push!(c_ex, e)
309 end
310 end
311 end
312 close(c)
313 if @isdefined(c_ex)
314 throw(c_ex)
315 end
316 nothing
317 end

370 # Capture interpolated variables in $() and move them to let-block
371 function _lift_one_interp!(e)
372 letargs = Any[] # store the new gensymed arguments
373 _lift_one_interp_helper(e, false, letargs) # Start out _not_ in a quote context (false)
374 letargs
375 end
376 _lift_one_interp_helper(v, _, _) = v
377 function _lift_one_interp_helper(expr::Expr, in_quote_context, letargs)
378 if expr.head === :$
379 if in_quote_context # This $ is simply interpolating out of the quote
380 # Now, we're out of the quote, so any _further_ $ is ours.
381 in_quote_context = false
382 else
383 newarg = gensym()
384 push!(letargs, :($(esc(newarg)) = $(esc(expr.args[1]))))
385 return newarg # Don't recurse into the lifted $() exprs
386 end
387 elseif expr.head === :quote
388 in_quote_context = true # Don't try to lift $ directly out of quotes
389 elseif expr.head === :macrocall
390 return expr # Don't recur into macro calls, since some other macros use $
391 end
392 for (i,e) in enumerate(expr.args)
393 expr.args[i] = _lift_one_interp_helper(e, in_quote_context, letargs)
394 end
395 expr
396 end

409 # runtime system hook called when a task finishes
410 function task_done_hook(t::Task)
411 # `finish_task` sets `sigatomic` before entering this function
412 err = istaskfailed(t)
413 result = task_result(t)
414 handled = false
415 if err
416 t.backtrace = catch_backtrace()
417 end
418
419 donenotify = t.donenotify
420 if isa(donenotify, ThreadSynchronizer)
421 lock(donenotify)
422 try
423 if !isempty(donenotify.waitq)
424 handled = true
425 notify(donenotify)
426 end
427 finally
428 unlock(donenotify)
429 end
430 end
431
432 if err && !handled && Threads.threadid() == 1
433 if isa(result, InterruptException) && isdefined(Base, :active_repl_backend) &&
434 active_repl_backend.backend_task.state === :runnable && isempty(Workqueue) &&
435 active_repl_backend.in_eval
436 throwto(active_repl_backend.backend_task, result) # this terminates the task
437 end
438 end
439 # Clear sigatomic before waiting
440 sigatomic_end()
441 try
442 wait() # this will not return
443 catch e
444 # If an InterruptException happens while blocked in the event loop, try handing
445 # the exception to the REPL task since the current task is done.
446 # issue #19467
447 if Threads.threadid() == 1 &&
448 isa(e, InterruptException) && isdefined(Base, :active_repl_backend) &&
449 active_repl_backend.backend_task.state === :runnable && isempty(Workqueue) &&
450 active_repl_backend.in_eval
451 throwto(active_repl_backend.backend_task, e)
452 else
453 rethrow()
454 end
455 end

525 function enq_work(t::Task)
526 (t.state === :runnable && t.queue === nothing) || error("schedule: Task not runnable")
527 tid = Threads.threadid(t)
528 # Note there are three reasons a Task might be put into a sticky queue
529 # even if t.sticky == false:
530 # 1. The Task's stack is currently being used by the scheduler for a certain thread.
531 # 2. There is only 1 thread.
532 # 3. The multiq is full (can be fixed by making it growable).
533 if t.sticky || tid != 0 || Threads.nthreads() == 1
534 if tid == 0
535 tid = Threads.threadid()
536 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
537 end
538 push!(Workqueues[tid], t)
539 else
540 tid = 0
541 if ccall(:jl_enqueue_task, Cint, (Any,), t) != 0
542 # if multiq is full, give to a random thread (TODO fix)
543 tid = mod(time_ns() % Int, Threads.nthreads()) + 1
544 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
545 push!(Workqueues[tid], t)
546 end
547 end
548 ccall(:jl_wakeup_thread, Cvoid, (Int16,), (tid - 1) % Int16)
549 return t
550 end

554 """
555 schedule(t::Task, [val]; error=false)
556
557 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
558 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
559
560 If a second argument `val` is provided, it will be passed to the task (via the return value of
561 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
562 the woken task.
563
564 # Examples
565 ```jldoctest
566 julia> a5() = sum(i for i in 1:1000);
567
568 julia> b = Task(a5);
569
570 julia> istaskstarted(b)
571 false
572
573 julia> schedule(b);
574
575 julia> yield();
576
577 julia> istaskstarted(b)
578 true
579
580 julia> istaskdone(b)
581 true
582 ```
583 """
584 function schedule(t::Task, @nospecialize(arg); error=false)
585 # schedule a task to be (re)started with the given value or exception
586 t.state === :runnable || Base.error("schedule: Task not runnable")
587 if error
588 t.queue === nothing || Base.list_deletefirst!(t.queue, t)
589 setfield!(t, :exception, arg)
590 else
591 t.queue === nothing || Base.error("schedule: Task not runnable")
592 setfield!(t, :result, arg)
593 end
594 enq_work(t)
595 return t
596 end

669 function ensure_rescheduled(othertask::Task)
670 ct = current_task()
671 W = Workqueues[Threads.threadid()]
672 if ct !== othertask && othertask.state === :runnable
673 # we failed to yield to othertask
674 # return it to the head of a queue to be retried later
675 tid = Threads.threadid(othertask)
676 Wother = tid == 0 ? W : Workqueues[tid]
677 pushfirst!(Wother, othertask)
678 end
679 # if the current task was queued,
680 # also need to return it to the runnable state
681 # before throwing an error
682 list_deletefirst!(W, ct)
683 nothing
684 end
