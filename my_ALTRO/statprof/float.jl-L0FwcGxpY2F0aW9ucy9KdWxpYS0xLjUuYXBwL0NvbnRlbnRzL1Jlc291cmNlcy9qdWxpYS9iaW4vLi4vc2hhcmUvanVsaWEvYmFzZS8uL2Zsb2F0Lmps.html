80 function Float64(x::UInt128)
81 x == 0 && return 0.0
82 n = 128-leading_zeros(x) # ndigits0z(x,2)
83 if n <= 53
84 y = ((x % UInt64) << (53-n)) & 0x000f_ffff_ffff_ffff
85 else
86 y = ((x >> (n-54)) % UInt64) & 0x001f_ffff_ffff_ffff # keep 1 extra bit
87 y = (y+1)>>1 # round, ties up (extra leading bit in case of next exponent)
88 y &= ~UInt64(trailing_zeros(x) == (n-54)) # fix last bit to round to even
89 end
90 d = ((n+1022) % UInt64) << 52
91 reinterpret(Float64, d + y)
92 end

140 # Float32 -> Float16 algorithm from:
141 # "Fast Half Float Conversion" by Jeroen van der Zijp
142 # ftp://ftp.fox-toolkit.org/pub/fasthalffloatconversion.pdf
143 #
144 # With adjustments for round-to-nearest, ties to even.
145 #
146 let _basetable = Vector{UInt16}(undef, 512),
147 _shifttable = Vector{UInt8}(undef, 512)
148 for i = 0:255
149 e = i - 127
150 if e < -25 # Very small numbers map to zero
151 _basetable[i|0x000+1] = 0x0000
152 _basetable[i|0x100+1] = 0x8000
153 _shifttable[i|0x000+1] = 25
154 _shifttable[i|0x100+1] = 25
155 elseif e < -14 # Small numbers map to denorms
156 _basetable[i|0x000+1] = 0x0000
157 _basetable[i|0x100+1] = 0x8000
158 _shifttable[i|0x000+1] = -e-1
159 _shifttable[i|0x100+1] = -e-1
160 elseif e <= 15 # Normal numbers just lose precision
161 _basetable[i|0x000+1] = ((e+15)<<10)
162 _basetable[i|0x100+1] = ((e+15)<<10) | 0x8000
163 _shifttable[i|0x000+1] = 13
164 _shifttable[i|0x100+1] = 13
165 elseif e < 128 # Large numbers map to Infinity
166 _basetable[i|0x000+1] = 0x7C00
167 _basetable[i|0x100+1] = 0xFC00
168 _shifttable[i|0x000+1] = 24
169 _shifttable[i|0x100+1] = 24
170 else # Infinity and NaN's stay Infinity and NaN's
171 _basetable[i|0x000+1] = 0x7C00
172 _basetable[i|0x100+1] = 0xFC00
173 _shifttable[i|0x000+1] = 13
174 _shifttable[i|0x100+1] = 13
175 end
176 end
177 global const shifttable = (_shifttable...,)
178 global const basetable = (_basetable...,)
179 end

181 function Float16(val::Float32)
182 f = reinterpret(UInt32, val)
183 if isnan(val)
184 t = 0x8000 ⊻ (0x8000 & ((f >> 0x10) % UInt16))
185 return reinterpret(Float16, t ⊻ ((f >> 0xd) % UInt16))
186 end
187 i = ((f & ~significand_mask(Float32)) >> significand_bits(Float32)) + 1
188 @inbounds sh = shifttable[i]
189 f &= significand_mask(Float32)
190 # If `val` is subnormal, the tables are set up to force the
191 # result to 0, so the significand has an implicit `1` in the
192 # cases we care about.
193 f |= significand_mask(Float32) + 0x1
194 @inbounds h = (basetable[i] + (f >> sh) & significand_mask(Float16)) % UInt16
195 # round
196 # NOTE: we maybe should ignore NaNs here, but the payload is
197 # getting truncated anyway so "rounding" it might not matter
198 nextbit = (f >> (sh-1)) & 1
199 if nextbit != 0 && (h & 0x7C00) != 0x7C00
200 # Round halfway to even or check lower bits
201 if h&1 == 1 || (f & ((1<<(sh-1))-1)) != 0
202 h += UInt16(1)
203 end
204 end
205 reinterpret(Float16, h)
206 end

208 function Float32(val::Float16)
209 local ival::UInt32 = reinterpret(UInt16, val)
210 local sign::UInt32 = (ival & 0x8000) >> 15
211 local exp::UInt32 = (ival & 0x7c00) >> 10
212 local sig::UInt32 = (ival & 0x3ff) >> 0
213 local ret::UInt32
214
215 if exp == 0
216 if sig == 0
217 sign = sign << 31
218 ret = sign | exp | sig
219 else
220 n_bit = 1
221 bit = 0x0200
222 while (bit & sig) == 0
223 n_bit = n_bit + 1
224 bit = bit >> 1
225 end
226 sign = sign << 31
227 exp = ((-14 - n_bit + 127) << 23) % UInt32
228 sig = ((sig & (~bit)) << n_bit) << (23 - 10)
229 ret = sign | exp | sig
230 end
231 elseif exp == 0x1f
232 if sig == 0 # Inf
233 if sign == 0
234 ret = 0x7f800000
235 else
236 ret = 0xff800000
237 end
238 else # NaN
239 ret = 0x7fc00000 | (sign<<31) | (sig<<(23-10))
240 end
241 else
242 sign = sign << 31
243 exp = ((exp - 15 + 127) << 23) % UInt32
244 sig = sig << (23 - 10)
245 ret = sign | exp | sig
246 end
247 return reinterpret(Float32, ret)
248 end

470 # Exact Float (Tf) vs Integer (Ti) comparisons
471 # Assumes:
472 # - typemax(Ti) == 2^n-1
473 # - typemax(Ti) can't be exactly represented by Tf:
474 # => Tf(typemax(Ti)) == 2^n or Inf
475 # - typemin(Ti) can be exactly represented by Tf
476 #
477 # 1. convert y::Ti to float fy::Tf
478 # 2. perform Tf comparison x vs fy
479 # 3. if x == fy, check if (1) resulted in rounding:
480 # a. convert fy back to Ti and compare with original y
481 # b. unsafe_convert undefined behaviour if fy == Tf(typemax(Ti))
482 # (but consequently x == fy > y)
483 for Ti in (Int64,UInt64,Int128,UInt128)
484 for Tf in (Float32,Float64)
485 @eval begin
486 function ==(x::$Tf, y::$Ti)
487 fy = ($Tf)(y)
488 (x == fy) & (fy != $(Tf(typemax(Ti)))) & (y == unsafe_trunc($Ti,fy))
489 end
490 ==(y::$Ti, x::$Tf) = x==y

603 """
604 nextfloat(x::AbstractFloat, n::Integer)
605
606 The result of `n` iterative applications of `nextfloat` to `x` if `n >= 0`, or `-n`
607 applications of `prevfloat` if `n < 0`.
608 """
609 function nextfloat(f::IEEEFloat, d::Integer)
610 F = typeof(f)
611 fumax = reinterpret(Unsigned, F(Inf))
612 U = typeof(fumax)
613
614 isnan(f) && return f
615 fi = reinterpret(Signed, f)
616 fneg = fi < 0
617 fu = unsigned(fi & typemax(fi))
618
619 dneg = d < 0
620 da = uabs(d)
621 if da > typemax(U)
622 fneg = dneg
623 fu = fumax
624 else
625 du = da % U
626 if fneg ⊻ dneg
627 if du > fu
628 fu = min(fumax, du - fu)
629 fneg = !fneg
630 else
631 fu = fu - du
632 end
633 else
634 if fumax - fu < du
635 fu = fumax
636 else
637 fu = fu + du
638 end
639 end
640 end
641 if fneg
642 fu |= sign_mask(F)
643 end
644 reinterpret(F, fu)
645 end

673 if Ti <: Unsigned || sizeof(Ti) < sizeof(Tf)
674 # Here `Tf(typemin(Ti))-1` is exact, so we can compare the lower-bound
675 # directly. `Tf(typemax(Ti))+1` is either always exactly representable, or
676 # rounded to `Inf` (e.g. when `Ti==UInt128 && Tf==Float32`).
677 @eval begin
678 function trunc(::Type{$Ti},x::$Tf)

688 else
689 throw(InexactError($(Expr(:quote,Ti.name.name)), $Ti, x))
690 end
691 end

696 # be rounded up. This assumes that `Tf(typemin(Ti)) > -Inf`, which is true for
697 # these types, but not for `Float16` or larger integer types.
698 @eval begin
699 function trunc(::Type{$Ti},x::$Tf)
700 if $(Tf(typemin(Ti))) <= x < $(Tf(typemax(Ti)))
701 return unsafe_trunc($Ti,x)
702 else
703 throw(InexactError(:trunc, $Ti, x))
704 end
705 end
706 function (::Type{$Ti})(x::$Tf)
707 if ($(Tf(typemin(Ti))) <= x < $(Tf(typemax(Ti)))) && (round(x, RoundToZero) == x)
708 return unsafe_trunc($Ti,x)
709 else

797
798 Return the *machine epsilon* of the floating point type `T` (`T = Float64` by
799 default). This is defined as the gap between 1 and the next largest value representable by
800 `typeof(one(T))`, and is equivalent to `eps(one(T))`. (Since `eps(T)` is a

802

805 julia> eps()

816 ```

820 """
821 eps(x::AbstractFloat)
822
823 Return the *unit in last place* (ulp) of `x`. This is the distance between consecutive
824 representable floating point values at `x`. In most cases, if the distance on either side
825 of `x` is different, then the larger of the two is taken, that is
826
827 eps(x) == max(x-prevfloat(x), nextfloat(x)-x)
828
829 The exceptions to this rule are the smallest and largest finite values
830 (e.g. `nextfloat(-Inf)` and `prevfloat(Inf)` for [`Float64`](@ref)), which round to the
831 smaller of the values.
832
833 The rationale for this behavior is that `eps` bounds the floating point rounding
834 error. Under the default `RoundNearest` rounding mode, if ``y`` is a real number and ``x``
835 is the nearest floating point number to ``y``, then
836
837 ```math
838 |y-x| \\leq \\operatorname{eps}(x)/2.
839 ```
840
841 # Examples
842 ```jldoctest
843 julia> eps(1.0)
844 2.220446049250313e-16
845
846 julia> eps(prevfloat(2.0))
847 2.220446049250313e-16
848
849 julia> eps(2.0)
850 4.440892098500626e-16
851
852 julia> x = prevfloat(Inf) # largest finite Float64
853 1.7976931348623157e308
854
855 julia> x + eps(x)/2 # rounds up
856 Inf
857
858 julia> x + prevfloat(eps(x)/2) # rounds down
859 1.7976931348623157e308
860 ```
861 """
862 eps(::AbstractFloat)
