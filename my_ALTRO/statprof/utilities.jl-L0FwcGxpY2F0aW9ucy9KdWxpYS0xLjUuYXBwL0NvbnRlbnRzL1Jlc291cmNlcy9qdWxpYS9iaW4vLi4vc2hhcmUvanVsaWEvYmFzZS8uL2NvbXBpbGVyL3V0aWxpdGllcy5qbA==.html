101 m = linfo.def::Method
102 c = nothing
103 if isdefined(m, :generator)

105 c = get_staged(linfo)
106 end

108 src = m.source

157 function argextype(@nospecialize(x), src, sptypes::Vector{Any}, slottypes::Vector{Any} = empty_slottypes)
158 if isa(x, Expr)
159 if x.head === :static_parameter
160 return sptypes[x.args[1]]
161 elseif x.head === :boundscheck
162 return Bool
163 elseif x.head === :copyast
164 return argextype(x.args[1], src, sptypes, slottypes)
165 end
166 @assert false "argextype only works on argument-position values"
167 elseif isa(x, SlotNumber)
168 return slottypes[(x::SlotNumber).id]
169 elseif isa(x, TypedSlot)
170 return (x::TypedSlot).typ
171 elseif isa(x, SSAValue)
172 return abstract_eval_ssavalue(x::SSAValue, src)
173 elseif isa(x, Argument)
174 return isa(src, IncrementalCompact) ? src.ir.argtypes[x.n] : src.argtypes[x.n]
175 elseif isa(x, QuoteNode)
176 return AbstractEvalConstant((x::QuoteNode).value)
177 elseif isa(x, GlobalRef)
178 return abstract_eval_global(x.mod, (x::GlobalRef).name)
179 elseif isa(x, PhiNode)
180 return Any
181 elseif isa(x, PiNode)
182 return x.typ
183 else
184 return AbstractEvalConstant(x)
185 end
186 end
