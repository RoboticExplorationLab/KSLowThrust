124 function sptypes_from_meth_instance(linfo::MethodInstance)
125 toplevel = !isa(linfo.def, Method)
126 if !toplevel && isempty(linfo.sparam_vals) && isa(linfo.def.sig, UnionAll)
127 # linfo is unspecialized
128 sp = Any[]
129 sig = linfo.def.sig
130 while isa(sig, UnionAll)
131 push!(sp, sig.var)
132 sig = sig.body
133 end
134 else
135 sp = collect(Any, linfo.sparam_vals)
136 end
137 for i = 1:length(sp)
138 v = sp[i]
139 if v isa TypeVar
140 fromArg = 0
141 # if this parameter came from arg::Type{T}, then `arg` is more precise than
142 # Type{T} where lb<:T<:ub
143 sig = linfo.def.sig
144 temp = sig
145 for j = 1:i-1
146 temp = temp.body
147 end
148 Pi = temp.var
149 while temp isa UnionAll
150 temp = temp.body
151 end
152 sigtypes = temp.parameters
153 for j = 1:length(sigtypes)
154 tj = sigtypes[j]
155 if isType(tj) && tj.parameters[1] === Pi
156 fromArg = j
157 break
158 end
159 end
160 if fromArg > 0
161 ty = fieldtype(linfo.specTypes, fromArg)
162 else
163 ub = v.ub
164 while ub isa TypeVar
165 ub = ub.ub
166 end
167 if has_free_typevars(ub)
168 ub = Any
169 end
170 lb = v.lb
171 while lb isa TypeVar
172 lb = lb.lb
173 end
174 if has_free_typevars(lb)
175 lb = Bottom
176 end
177 if Any <: ub && lb <: Bottom
178 ty = Any
179 else
180 tv = TypeVar(v.name, lb, ub)
181 ty = UnionAll(tv, Type{tv})
182 end
183 end
184 else
185 ty = Const(v)
186 end
187 sp[i] = ty
188 end
189 return sp
190 end

205 function record_ssa_assign(ssa_id::Int, @nospecialize(new), frame::InferenceState)
206 old = frame.src.ssavaluetypes[ssa_id]
207 if old === NOT_FOUND || !(new ⊑ old)
208 # typically, we expect that old ⊑ new (that output information only
209 # gets less precise with worse input information), but to actually
210 # guarantee convergence we need to use tmerge here to ensure that is true
211 frame.src.ssavaluetypes[ssa_id] = old === NOT_FOUND ? new : tmerge(old, new)
212 W = frame.ip
213 s = frame.stmt_types
214 for r in frame.ssavalue_uses[ssa_id]
215 if s[r] !== nothing # s[r] === nothing => unreached statement
216 if r < frame.pc´´
217 frame.pc´´ = r
218 end
219 push!(W, r)
220 end
221 end
222 end
223 nothing
224 end

255 function poison_callstack(infstate::InferenceState, topmost::InferenceState, poison_topmost::Bool)
256 poison_topmost && (topmost = topmost.parent)
257 while !(infstate === topmost)
258 if call_result_unused(infstate)
259 # If we won't propagate the result any further (since it's typically unused),
260 # it's OK that we keep and cache the "limited" result in the parents
261 # (non-typically, this means that we lose the ability to detect a guaranteed StackOverflow in some cases)
262 # TODO: we might be able to halt progress much more strongly here,
263 # since now we know we won't be able to keep anything much that we learned.
264 # We were mainly only here to compute the calling convention return type,
265 # but in most situations now, we are unlikely to be able to use that information.
266 break
267 end
268 infstate.limited = true
269 for infstate_cycle in infstate.callers_in_cycle
270 infstate_cycle.limited = true
271 end
272 infstate = infstate.parent
273 infstate === nothing && return
274 end
