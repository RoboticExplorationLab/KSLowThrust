412 # Walk through `linfo`'s upstream call chain, starting at `parent`. If a parent
413 # frame matching `linfo` is encountered, then there is a cycle in the call graph
414 # (i.e. `linfo` is a descendant callee of itself). Upon encountering this cycle,
415 # we "resolve" it by merging the call chain, which entails unioning each intermediary
416 # frame's `callers_in_cycle` field and adding the appropriate backedges. Finally,
417 # we return `linfo`'s pre-existing frame. If no cycles are found, `nothing` is
418 # returned instead.
419 function resolve_call_cycle!(linfo::MethodInstance, parent::InferenceState)
420 frame = parent
421 uncached = false
422 limited = false
423 while isa(frame, InferenceState)
424 uncached |= !frame.cached # ensure we never add an uncached frame to a cycle
425 limited |= frame.limited
426 if frame.linfo === linfo
427 if uncached
428 # our attempt to speculate into a constant call lead to an undesired self-cycle
429 # that cannot be converged: poison our call-stack (up to the discovered duplicate frame)
430 # with the limited flag and abort (set return type to Any) now
431 poison_callstack(parent, frame, false)
432 return true
433 end
434 merge_call_chain!(parent, frame, frame, limited)
435 return frame
436 end
437 for caller in frame.callers_in_cycle
438 if caller.linfo === linfo
439 if uncached
440 poison_callstack(parent, frame, false)
441 return true
442 end
443 merge_call_chain!(parent, frame, caller, limited)
444 return caller
445 end
446 end
447 frame = frame.parent
448 end
449 return false
