264 function is_pure_intrinsic_infer(f::IntrinsicFunction)
265 return !(f === Intrinsics.pointerref || # this one is volatile
266 f === Intrinsics.pointerset || # this one is never effect-free
267 f === Intrinsics.llvmcall || # this one is never effect-free
268 f === Intrinsics.arraylen || # this one is volatile
269 f === Intrinsics.sqrt_llvm || # this one may differ at runtime (by a few ulps)
270 f === Intrinsics.sqrt_llvm_fast || # this one may differ at runtime (by a few ulps)
271 f === Intrinsics.cglobal) # cglobal lookup answer changes at runtime
272 end
